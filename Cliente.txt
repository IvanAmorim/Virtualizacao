interface vlan 3

#Set User / Password
local-user admin
	service-type telnet
	authorization-attribute level 3
	password simple admin
quit

#HTTP Access
ip http enable 
ip http port 80

#Telnet Access
telnet server enable
user-interface vty 0 5
authentication-mode password
set authentication password simple admin
user privilege level 3
